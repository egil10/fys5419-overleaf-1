# Solve for lambda in [0, 1]
lambdas = np.linspace(0, 1, 101)
eigvals = []
eigvecs = []

for l in lambdas:
    H = H0 + l * HI
    w, v = np.linalg.eigh(H)
    eigvals.append(w)
    eigvecs.append(v)

eigvals = np.array(eigvals)
eigvecs = np.array(eigvecs)
\end{lstlisting}
